```typescript
const HeroRenderer: React.FC<{ props: any }> = ({ props }) => {
  const backgroundStyle: React.CSSProperties = {
    height: props.height || 400,
    display: 'flex',
    flexDirection: 'column',
    alignItems: 'center',
    justifyContent: 'center',
    textAlign: props.textAlign || 'center',
    color: '#fff',
    padding: '40px 20px',
  };

  // 背景图片处理
  if (props.backgroundImage) {
    backgroundStyle.backgroundImage = `url(${props.backgroundImage})`;
    
    // 填充模式
    switch (props.backgroundFit) {
      case 'cover': backgroundStyle.backgroundSize = 'cover'; break;
      case 'contain': backgroundStyle.backgroundSize = 'contain'; break;
      case 'fill': backgroundStyle.backgroundSize = '100% 100%'; break;
      case 'repeat': 
        backgroundStyle.backgroundSize = 'auto';
        backgroundStyle.backgroundRepeat = 'repeat';
        break;
      default: backgroundStyle.backgroundSize = 'cover';
    }
    
    // 位置设置
    switch (props.backgroundPosition) {
      case 'center': backgroundStyle.backgroundPosition = 'center'; break;
      case 'top': backgroundStyle.backgroundPosition = 'center top'; break;
      case 'bottom': backgroundStyle.backgroundPosition = 'center bottom'; break;
      case 'left': backgroundStyle.backgroundPosition = 'left center'; break;
      case 'right': backgroundStyle.backgroundPosition = 'right center'; break;
      default: backgroundStyle.backgroundPosition = 'center';
    }
  } else {
    // 降级到背景色
    backgroundStyle.background = props.backgroundColor || '#1677ff';
  }

  return <div style={backgroundStyle}>...</div>;
};
```

#### 渲染器实现
**文件**: `company_cms_project/frontend/src/components/ComponentRenderer.tsx`

**核心逻辑**:
```typescript
const ButtonRenderer: React.FC<{ props: any }> = ({ props }) => {
  const navigate = useNavigate();
  
  // 计算链接地址
  const getLinkUrl = (): string | null => {
    const linkType = props.linkType || 'none';
    const linkValue = props.linkValue;
    
    switch (linkType) {
      case 'home':
        return '/';
      case 'page':
        return linkValue ? `/${linkValue}` : null;
      case 'url':
        return linkValue || null;
      default:
        return null;
    }
  };
  
  // 处理点击
  const handleClick = () => {
    const url = getLinkUrl();
    if (!url) return;
    
    if (props.linkType === 'url') {
      window.open(url, '_blank');  // 外部链接新窗口
    } else {
      navigate(url);  // 内部导航
    }
  };
  
  const linkUrl = getLinkUrl();
  
  // 内部链接使用 Link 组件
  if (linkUrl && props.linkType !== 'url') {
    return (
      <div style={{ padding: '16px', textAlign: 'center' }}>
        <Link to={linkUrl} style={{ textDecoration: 'none' }}>
          <Button type={buttonType} size={size} block={block}>
            {text}
          </Button>
        </Link>
      </div>
    );
  }
  
  // 无链接或外部链接
  return (
    <div style={{ padding: '16px', textAlign: 'center' }}>
      <Button onClick={handleClick}>{text}</Button>
    </div>
  );
};
```

## 🔧 技术实现细节

### 1. 复用现有组件
- **ImagePicker 组件**: 直接复用于背景图片选择，已支持：
  - 媒体库浏览
  - 拖拽上传
  - URL 输入
  - 自动压缩和缩略图生成

### 2. 响应式 CSS 属性映射
| 填充模式 | CSS `background-size` | 适用场景 |
|---------|----------------------|---------|
| Cover   | `cover`              | 全屏背景图，保持比例裁剪 |
| Contain | `contain`            | 完整显示图片，可能留白 |
| Fill    | `100% 100%`          | 拉伸填满，可能变形 |
| Repeat  | `auto` + `repeat`    | 平铺图案背景 |

### 3. 路由导航策略
- **内部链接**: 使用 React Router 的 `<Link>` 组件，SPA 无刷新跳转
- **外部链接**: 使用 `window.open()` 新窗口打开，提升用户体验
- **首页快捷**: 硬编码路径 `/`，快速配置

## 📝 使用说明

### 横幅大图背景图片配置
1. 进入页面编辑器 (`/admin/page-editor`)
2. 添加或编辑"横幅大图"组件
3. 在右侧属性面板找到"背景图片"配置项
4. 点击选择器图标：
   - 从媒体库选择已有图片
   - 上传新图片
   - 输入外部图片 URL
5. 选择填充模式（推荐 Cover）
6. 选择图片位置（推荐居中）
7. 实时预览效果并保存

### 按钮链接配置
1. 进入页面编辑器
2. 添加或编辑"按钮"组件
3. 在右侧属性面板：
   - 选择"链接类型"
   - 如选"站内页面"，从下拉列表选择目标页面
   - 如选"外部链接"，输入完整 URL（含 https://）
4. 保存并预览
